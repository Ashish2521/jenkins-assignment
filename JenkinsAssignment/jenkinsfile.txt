def builtDockerImage
pipeline{
agent any
stages {
stage('Clone repo') {
            steps {
               git url: 'https://github.com/Ashish2521/jenkins-assignment.git', branch: 'main'
            }
        }
       stage('Build image') {
            steps {
                script {
                    builtDockerImage = docker.build "ashish2521/dockerjenkins:v1.0"
                }
            }
        }
        stage('Push image') {
            steps {
                script {
                        docker.withRegistry('', 'dockerhub-cred-manank') {
                            builtDockerImage.push()
                        }
                }
            }
        }
    }
}
